18. JS Bubble Sort

Sorts array in descending order.

Output 
function bubbleSort(arr) {
  for(let i=0; i<arr.length; i++){
    for(let j=0; j<arr.length-i-1; j++){
      if(arr[j] < arr[j+1]) { // Descending
         let temp = arr[j]; arr[j] = arr[j+1]; arr[j+1] = temp;
      }
    }
  }
}
